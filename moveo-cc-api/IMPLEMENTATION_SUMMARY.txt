const io = require('socket.io-client');
const socket = io('http://localhost:3001');

// Subscribe to vehicle

  console.log('Subscribed:', res.success);

  });
});

  vehicleId: 'truck_001',
  timestamp: Date.now(),
  latitude: 40.7128,
  longitude: -74.0060,
  speed: 55,
  bearing: 270
});

// Get stats
socket.emit('request:stats', {}, (stats) => {
  console.log(`Connected clients: ${stats.connectedClients}`);
  console.log(`Active vehicles: ${stats.activeVehicles}`);
});
